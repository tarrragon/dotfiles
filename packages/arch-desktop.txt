# Arch desktop 層 — 圖形桌面（Hyprland + rice + 字型），只在有畫面的機器裝
# 這層是平台分歧最大的一層：Hyprland 生態只存在於 Linux（且以 Arch 打包最全）

# compositor + 終端機
hyprland          # Wayland compositor（Aquamarine 渲染後端、v0.55+）
foot              # Wayland 原生終端機、輕量

# 字型（字型的可用集合在 process 啟動時決定、開機早期就要裝好）
ttf-meslo-nerd    # Powerlevel10k 推薦字型，字族名 "MesloLGS Nerd Font"
noto-fonts-cjk    # CJK fallback（Nerd Font 不含中日韓、缺這支通知/標題會豆腐）

# rice（step 2 已實測、全在 VM 正常 render）
waybar            # 狀態列（模組對缺硬體自動退化）
wofi              # 應用啟動器
mako              # 通知 daemon（只負責顯示）
libnotify         # notify-send + 應用程式的通知發送端
hyprlock          # 鎖屏（ext-session-lock 協議、殺 process 不等於解鎖）
hyprpaper         # 靜態桌布
xdg-desktop-portal-hyprland  # 螢幕分享 / 檔案選擇 portal
wl-clipboard      # 剪貼簿（wl-copy / wl-paste）
grim              # 截圖
slurp             # 截圖區域框選
# 截圖 keybind 記得帶 --type image/png（無 xdg-utils 時 wl-copy 會誤標 MIME）

# 音訊（pipewire 常被依賴鏈拉進來、但 session manager 不會自動跟進——
# 沒 wireplumber 就沒有任何 sink、應用無聲且不報錯；權威判讀 wpctl status 看 Sinks）
wireplumber       # pipewire session manager（建立 sink/source 的那個）
pipewire-pulse    # PulseAudio 相容層（多數 GUI 應用走 pulse API）
pipewire-alsa     # ALSA 相容層

# 檔案管理（VM 實測：thunar 視窗正常、gvfs 側欄掛載正常）
thunar            # GTK 檔案管理員（不拖桌面環境）
thunar-volman     # 隨身碟/外接裝置自動掛載
thunar-archive-plugin  # 右鍵解壓縮（配 file-roller）
file-roller       # 壓縮檔前端
gvfs              # 掛載/垃圾桶/網路（側欄 Devices/Network/Trash 靠它）
tumbler           # 縮圖服務

# 瀏覽器（ALARM aarch64 repo 都有官方建置）
firefox
chromium

# 媒體播放（Arch 把 VLC 解碼器拆包——本體只有 UI、缺 plugin 播 h264 直接報
# codec not supported；首跑會彈 Privacy/Network Access 同意對話框、屬預期互動）
vlc
vlc-plugin-ffmpeg # 解碼器（沒有它 = 只能開不能播）
ffmpeg            # CLI 轉檔 / 生測試素材
# Spotify 等 DRM 串流（aarch64）：官方 client 只發 x86_64 deb（spotify-launcher 會報
# no packages for your cpu's architecture）、走 Web Player + Widevine。CDM 手動裝：
#   git clone https://github.com/AsahiLinux/widevine-installer
#   sudo env LIBDIR=/usr/lib CHROME_WIDEVINE_BASE=/usr/lib/chromium \
#     MOZ_PREF_BASE=/usr/lib/firefox/defaults/pref ./widevine-installer
# 需 squashfs-tools；裝完 chromium 要「重啟一次」CDM 才註冊（hint 檔時序）

# 圖形檔案管理員（選 Thunar：桌面無關、輕量標準；在裸 Hyprland 只新裝 8 個相依）
# 側欄的掛載/垃圾桶/網路、縮圖不是 Thunar 內建、各是一個額外套件，明確補上：
thunar                # XFCE 出身、但只拖 xfce4 基礎庫、不裝 XFCE 桌面本身
gvfs                  # 側欄 Devices/Network/Trash 的後端（掛載/垃圾桶/網芳）
tumbler               # 縮圖服務（圖片）
ffmpegthumbnailer     # 影片縮圖（tumbler 的 optional dep）
