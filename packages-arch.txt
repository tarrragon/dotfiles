# Arch / Hyprland 桌面套件清單
# install.sh 會過濾 # 註解與空行、其餘餵給 pacman -S --needed

# 前置（不在此清單、需先手動裝，因為 install.sh 自身要靠它）：

# === 第一步：最小可跑（Hyprland 啟動 + 終端機）===
hyprland          # Wayland compositor（Aquamarine 渲染後端、v0.55+）
foot              # Wayland 原生終端機、輕量、GPU 加速

# === 開發環境：shell rice + 工具鏈 ===

# HTTPS / git clone 的信任根（oh-my-zsh / p10k / Claude Code 安裝都要 HTTPS）
ca-certificates

# shell rice 支援
autojump          # 目錄跳轉（oh-my-zsh autojump plugin 會 source 它）
ttf-meslo-nerd    # Powerlevel10k 推薦字型，字族名 "MesloLGS Nerd Font"
noto-fonts-cjk    # CJK fallback 字型（Nerd Font 不含中日韓、缺這支通知/標題會豆腐）
# 註：oh-my-zsh / powerlevel10k / zsh-autosuggestions / zsh-syntax-highlighting
#     由 install.sh git clone 進 ~/.oh-my-zsh/custom（對齊 .zshrc 的 OMZ plugin 機制）

# stowed config 但 base 沒裝的 binary
broot             # 互動式目錄樹瀏覽器
btop              # 系統資源監控 TUI
zellij            # 終端機多工器（取代 tmux、SSH 無人值守任務靠它）

# 核心 CLI
curl              # CC 原生 installer 與一般下載都要
github-cli        # gh（PR / issue / auth 操作）
fzf               # 模糊搜尋（搭配 Ctrl+R 歷史搜尋）

# 常用 CLI
ripgrep           # rg（grep 替代、速度快、預設 .gitignore 過濾）
fd                # find 替代、語法簡潔
bat               # cat 替代、帶語法高亮
lazygit           # Git 操作 TUI
git-delta         # diff 語法高亮 pager（.gitconfig core.pager 指向它）
yazi              # 終端機檔案管理器
tig               # Git 歷史 TUI
ranger            # 終端機檔案管理器（三欄式）

# Claude Code 的 runtime（CC 由 install.sh 用原生 installer 裝進 ~/.local/bin）
nodejs
npm

# === Hyprland rice（step 2 已驗證、全在 VM 正常 render）===
waybar            # 狀態列（JSON config + CSS、模組對缺硬體自動退化）
wofi              # Wayland 原生應用啟動器
mako              # 通知 daemon（只負責顯示、產生通知靠 libnotify）
libnotify         # notify-send + 應用程式的通知發送端
hyprlock          # 鎖屏（ext-session-lock 協議、殺 process 不等於解鎖）
hyprpaper         # 靜態桌布
xdg-desktop-portal-hyprland  # 螢幕分享 / 檔案選擇 portal
wl-clipboard      # Wayland 剪貼簿存取（wl-copy / wl-paste）
grim              # Wayland 截圖工具
slurp             # 截圖區域框選工具
# 截圖 keybind 記得帶 --type image/png（無 xdg-utils 時 wl-copy 會誤標 MIME）
